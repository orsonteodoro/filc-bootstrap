# Instructions for creating a Debian guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts for correctness quickly and glibc build.

# On the Gentoo host do

mkdir -p ~/qemu-debian
cd ~/qemu-debian

# Download the Debian 12 netinst ISO (current as of 2026)
wget https://cdimage.debian.org/cdimage/archive/12.10.0/amd64/iso-cd/debian-12.10.0-amd64-netinst.iso

# Create virtual disk
qemu-img create -f qcow2 debian.qcow2 80G

# The following configurations are assumed:
#
# SMOOTHER=1:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 4 Core, 16 GiB total memory for guest
# SMOOTHER=2:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 2 Core, 16 GiB total memory for guest
# SMOOTHER=4:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 1 Core, 16 GiB total memory for guest
#
# We let the host handle the swapfile issue for the heavy linking memory requirements.
# The host needs at least 16 GiB total memory, because linking can take 13 GiB.
# HOST_TOTAL_MEMORY = HOST_RAM + HOST_SWAP
#

# Contents of ~/qemu-debian/setup.sh (to boot the installer)
#!/bin/bash
SMOOTHER=4 # Increasing it makes the host smoother but increases build time
HOST_NCORES=4
GUEST_VM_NCORES=$(( ${HOST_NCORES} / ${SMOOTHER} )) # This should be HOST_NCORES / 2 for smoother multi tasking on host and reduced thrashing
GUEST_VM_TOTAL_MEMORY=$(( 16 * 1024 ))
(( ${GUEST_VM_NCORES} <= 0 )) && GUEST_VM_NCORES=1
qemu-system-x86_64 \
  -enable-kvm \
  -m ${GUEST_VM_TOTAL_MEMORY} \
  -cpu host \
  -smp ${GUEST_VM_NCORES} \
  -nic user,model=virtio \
  -drive file=debian.qcow2,format=qcow2,if=virtio \
  -cdrom debian-12.10.0-amd64-netinst.iso \
  -boot once=d \
  -display gtk \
  -vga virtio \
  -usb -device usb-tablet \
  "$@"

chmod +x setup.sh
./setup.sh

# Install debian.
# For most options, use the defaults except swap space.
# Delete the swap parition.  We will use a swap file instead.
# Disable the graphical install options.
# No swap partition or swap file in guest

# Install is done.

# Let the installer restart.
# When it reaches Grub, shutdown the VM.

# Outside in the host do
