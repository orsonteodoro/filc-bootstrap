# Instructions for creating a Debian guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts for correctness quickly and glibc build.

# On the Gentoo host do

mkdir -p ~/qemu-debian
cd ~/qemu-debian

# Download the Debian 12 netinst ISO (current as of 2026)
wget https://cdimage.debian.org/cdimage/archive/12.10.0/amd64/iso-cd/debian-12.10.0-amd64-netinst.iso

# Create virtual disk
qemu-img create -f raw debian.raw 80G

# Do not change the -m values.  It may increase linking time.

# Contents of ~/qemu-debian/setup.sh (to boot the installer)
#!/bin/bash
qemu-system-x86_64 \
  -enable-kvm \
  -m 8192 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=debian.raw,format=raw,if=virtio,cache=none \
  -cdrom debian-12.10.0-amd64-netinst.iso \
  -boot once=d \
  -display gtk \
  -vga virtio \
  -usb -device usb-tablet \
  "$@"

chmod +x setup.sh
./setup.sh

# Install debian as "Install".
# For most options, use the defaults except swap space.
# Delete the swap parition.  We will use a swap file instead.
# Disable the graphical install options.
# Manually partition disks as follows
#   vda
#     - 84.9 GB Primary xfs boot flag on placed at the beginning of the drive for /
#     - 1.0 GB Logical swap area
# Install grub in /dev/vda

# Install grub

# Install is done.

# Let the installer restart.
# When it reaches Grub, shutdown the VM.

# Outside in the host do

# Contents of ~/qemu-debian/setup2.sh (to fetch filc-bootstrap)
#!/bin/bash
qemu-system-x86_64 \
  -enable-kvm \
  -m 8192 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=debian.raw,format=raw,if=virtio,cache=none \
  -display gtk \
  -vga virtio \
  -usb -device usb-tablet \
  "$@"
chmod +x setup2.sh
./setup2.sh

# In the Debian guest do

# Create an 12 GiB swap file to prevent OOM and random failures
fallocate -l 12G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile

# Make it permanent across reboots
echo '/swapfile none swap sw 0 0' >> /etc/fstab

# * Do not change the swap recipe.  It is to restore back the performant config before linker completion time issues.

cd ~/
apt update && apt upgrade -y
apt install -y git

# For developers/testers add screen for inspecting errors
apk install screen
# Increase scrollback
echo "defscrollback 50000" > ~/.screenrc
screen
# Use CTRL a [ to scroll back and to report/fix errors

# Copy the filc-bootstrap repo into the VM
git clone https://github.com/orsonteodoro/filc-bootstrap.git
cd filc-bootstrap

# Download dependencies and filc-bootstrap with
chmod +x setup-debian-guest-filc-test.sh
./setup-debian-guest-filc-test.sh

# Run the test without a web browser or without a memory hungry program while bootstrapping.
cd filc-bootstrap
./bootstrap.sh --test-debian --fresh
