# Instructions for creating a Debian guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts for correctness quickly and glibc build.

# On the Gentoo host do

mkdir -p ~/qemu-debian
cd ~/qemu-debian

# Download the Debian 12 netinst ISO (current as of 2026)
wget https://cdimage.debian.org/cdimage/archive/12.10.0/amd64/iso-cd/debian-12.10.0-amd64-netinst.iso

# Create virtual disk
qemu-img create -f qcow2 debian.qcow2 40G

#
# The GUEST_VM_RAM should be RAM_HOST/2 or RAM_HOST/4 to prevent swapping.
# The GUEST_VM_SWAP should be 16384 - RAM_HOST/2
# DO NOT USE GUEST_VM_RAM == RAM_HOST.
#
# Use one of the following recommened combos:
# 8 GiB host, 4 GiB RAM, 12 Swap, 16 GiB total VM Memory
# 16 GiB host, 8 GiB RAM, 8 Swap, 16 GiB total VM Memory
# 32 GiB host, 16 GiB RAM, 0 Swap, 16 GiB total VM Memory
#

# Contents of ~/qemu-debian/setup.sh (to boot the installer)
qemu-system-x86_64 \
  -enable-kvm \
  -m 4096 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=debian.qcow2,format=qcow2,if=virtio \
  -cdrom debian-12.10.0-amd64-netinst.iso \
  -boot once=d \
  -display gtk \
  -vga virtio \
  -usb -device usb-tablet

chmod +x setup.sh
./setup.sh

# Install debian
# Install is done

# 1. Create an 12 GiB swap file
fallocate -l 12G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile

# 2. Make it permanent across reboots
echo '/swapfile none swap sw 0 0' >> /etc/fstab

sync
# Shutdown vm

# Outside in the host do

# Verify 16 GiB total memory
free

# Contents of ~/qemu-debian/setup2.sh (to fetch filc-bootstrap)
# The total memory needs to be 16 GiB for possibly worst case scenario for linking at maybe 13 GiB total memory.
qemu-system-x86_64 \
  -enable-kvm \
  -m 16384 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=debian.qcow2,format=qcow2,if=virtio \
  -display gtk \
  -vga virtio \
  -usb -device usb-tablet \
  -name "Debian12-FilC-Test" \
  "$@"
chmod +x setup2.sh
./setup2.sh

# In the Debian guest:
cd ~/
sudo apt update && sudo apt upgrade -y
sudo apt install -y git

# Download dependencies and filc-bootstrap with
chmod +x setup-debian-guest-filc-test.sh
./setup-debian-guest-filc-test.sh

# Run the test
cd filc-bootstrap
./bootstrap.sh --test-debian --fresh
