# Instructions for creating a Alpine guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts correctness quickly and musl build.

# WARNING:  Currently this testing case is on hold until the llvm annotations dependency can be resolved.
# Use the Debian testing instructions instead.

# Outside in the host do

mkdir -p ~/qemu-alpine
cd ~/qemu-alpine

# Get the latest stable Alpine
wget https://dl-cdn.alpinelinux.org/alpine/v3.23/releases/x86_64/alpine-standard-3.23.0-x86_64.iso

cd ~/qemu-alpine
qemu-img create -f qcow2 alpine.qcow2 80G

# The following configurations are assumed:
#
# SMOOTHER=1:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 4 Core, 16 GiB total memory for guest
# SMOOTHER=2:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 2 Core, 16 GiB total memory for guest
# SMOOTHER=4:  Assumes 4 Core, 8 GiB RAM, 24 GiB swap for host; 1 Core, 16 GiB total memory for guest
#
# We let the host handle the swapfile issue for the heavy linking memory requirements.
# The host needs at least 16 GiB total memory, because linking can take 13 GiB.
# HOST_TOTAL_MEMORY = HOST_RAM + HOST_SWAP
#

# Contents of ~/qemu-alpine/setup.sh for distro install:
#!/bin/bash
SMOOTHER=4 # Increasing	it makes the host smoother but increases build time
HOST_NCORES=4
GUEST_VM_NCORES=$(( ${HOST_NCORES} / ${SMOOTHER} )) # This should be HOST_NCORES / 2 for smoother multi tasking on host and reduced thrashing
GUEST_VM_TOTAL_MEMORY=$(( 16 * 1024 ))
(( ${GUEST_VM_NCORES} <= 0 )) && GUEST_VM_NCORES=1
qemu-system-x86_64 \
  -enable-kvm \
  -m ${GUEST_VM_TOTAL_MEMORY} \
  -cpu host \
  -smp ${GUEST_VM_NCORES} \
  -nic user,model=virtio \
  -drive file=alpine.qcow2,format=qcow2,if=virtio \
  -cdrom alpine-standard-3.23.0-x86_64.iso \
  -boot once=d \
  -display gtk \
  -vga virtio \
  "$@"
chmod +x ~/qemu-alpine/setup.sh
./setup.sh

# In the Alpine guest do

# login as root
setup-alpine
# Most options you just press enter.  The ones to worry about are
# Password → of your choosing
# Keyboard layout → usually us
# Disk → choose vda then choose sys mode
# No swap partition or swap file in guest

# Install is done.

sync
# Shutdown the VM.

# Outside in the host do

# In the Alpine guest do

# Verify 16 GiB total memory
free

# Update available packages list and tools required to get filc-bootstrap
cd ~/
apk update
apk add git bash

# For developers/testers add screen for inspecting errors
apk add screen
# Increase scrollback
echo "defscrollback 50000" > ~/.screenrc
screen
# Use CTRL a [ to scroll back and to report/fix errors

git clone https://github.com/orsonteodoro/filc-bootstrap.git
cd filc-bootstrap

chmod +x ./setup-alpine-guest-filc-test.sh
./setup-alpine-guest-filc-test.sh
./bootstrap.sh --test-alpine
