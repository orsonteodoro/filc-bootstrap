# Instructions for creating a Alpine guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts correctness quickly and musl build.

# WARNING:  Currently this testing case is on hold until the llvm annotations dependency can be resolved.
# Use the Debian testing instructions instead.

# Outside in the host do

mkdir -p ~/qemu-alpine
cd ~/qemu-alpine

# Get the latest stable Alpine
wget https://dl-cdn.alpinelinux.org/alpine/v3.23/releases/x86_64/alpine-standard-3.23.0-x86_64.iso

cd ~/qemu-alpine
qemu-img create -f qcow2 alpine.qcow2 80G

#
# The GUEST_VM_RAM should be RAM_HOST/2 or RAM_HOST/4 to prevent swapping.
# The GUEST_VM_SWAP should be 16384 - RAM_HOST/2
# DO NOT USE GUEST_VM_RAM == RAM_HOST.  It will cause severe swapping.
#
# Use one of the following recommened combos:
# 8 GiB host, 4 GiB RAM, 12 Swap, 16 GiB total VM Memory
# 16 GiB host, 8 GiB RAM, 8 Swap, 16 GiB total VM Memory
# 32 GiB host, 16 GiB RAM, 0 Swap, 16 GiB total VM Memory
#
# Change the -m arg for the VM based on `GUEST_VM_RAM=4; echo $((1024 * GUEST_VM_RAM))`
#

# Contents of ~/qemu-alpine/setup.sh for distro install:
#!/bin/bash
GUEST_VM_NCORES=2 # This should be HOST_NCORES / 2 for smoother multi tasking on host and reduced thrashing
GUEST_VM_RAM=4096
qemu-system-x86_64 \
  -enable-kvm \
  -m ${GUEST_VM_RAM} \
  -cpu host \
  -smp ${GUEST_VM_NCORES} \
  -nic user,model=virtio \
  -drive file=alpine.qcow2,format=qcow2,if=virtio \
  -cdrom alpine-standard-3.23.0-x86_64.iso \
  -boot once=d \
  -display gtk \
  -vga virtio
chmod +x ~/qemu-alpine/setup.sh
./setup.sh

# In the Alpine guest do

# login as root
setup-alpine
# Most options you just press enter.  The ones to worry about are
# Password → of your choosing
# Keyboard layout → usually us
# Disk → choose vda then choose sys mode

# Install is done.

sync
# Shutdown the VM.

# Outside in the host do

# Contents of ~/qemu-alpine/setup2.sh to boot without ISO image:
#!/bin/bash
# The total memory needs to be 16 GiB for possibly worst case scenario for
# linking.
GUEST_VM_NCORES=2 # This should be HOST_NCORES / 2 for smoother multi tasking on host and reduced thrashing
GUEST_VM_RAM=4096
qemu-system-x86_64 \
  -enable-kvm \
  -m ${GUEST_VM_RAM} \
  -cpu host \
  -smp ${GUEST_VM_NCORES} \
  -nic user,model=virtio \
  -drive file=alpine.qcow2,format=qcow2,if=virtio \
  -display gtk \
  -vga virtio
chmod +x ~/qemu-alpine/setup2.sh
./setup2.sh

# In the Alpine guest do

# Create a 12 GiB swap file
GUEST_VM_TOTAL_MEMORY=16
GUEST_VM_RAM=4
GUEST_VM_SWAP=$(( ${GUEST_VM_TOTAL_MEMORY} - ${GUEST_VM_RAM} ))
fallocate -l ${GUEST_VM_SWAP}G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile

# Make it permanent across reboots
echo '/swapfile none swap sw 0 0' >> /etc/fstab

# Verify 16 GiB total memory
free

# Update available packages list and tools required to get filc-bootstrap
cd ~/
apk update
apk add git bash

# For developers/testers add screen for inspecting errors
apk add screen
# Increase scrollback
echo "defscrollback 50000" > ~/.screenrc
screen
# Use CTRL a [ to scroll back and to report/fix errors

# Copy the filc-bootstrap repo into the VM
git clone https://github.com/orsonteodoro/filc-bootstrap.git
cd filc-bootstrap

chmod +x ./setup-alpine-guest-filc-test.sh
./setup-alpine-guest-filc-test.sh
./bootstrap.sh --test-alpine
