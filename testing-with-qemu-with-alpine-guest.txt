# Instructions for creating a Alpine guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts correctness quickly and musl build.

# Outside in the host do

mkdir -p ~/qemu-alpine
cd ~/qemu-alpine
wget https://dl-cdn.alpinelinux.org/alpine/v3.23/releases/x86_64/alpine-standard-3.23.0-x86_64.iso

cd ~/qemu-alpine
qemu-img create -f qcow2 alpine.qcow2 40G

# Contents of ~/qemu-alpine/setup.sh for distro install:
qemu-system-x86_64 \
  -enable-kvm \
  -m 4096 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=alpine.qcow2,format=qcow2,if=virtio \
  -cdrom alpine-standard-3.23.0-x86_64.iso \
  -boot once=d \
  -display gtk \
  -vga virtio
chmod +x ~/qemu-alpine/setup.sh
./setup.sh

# In the Alpine guest do

# login as root
setup-alpine
# Most options you just press enter.  The ones to worry about are
# Password → of your choosing
# Keyboard layout → usually us
# Disk → choose vda then choose sys mode

sync
# Shutdown vm

# Outside in the host do

# Contents of ~/qemu-alpine/setup2.sh to boot without ISO image:
qemu-system-x86_64 \
  -enable-kvm \
  -m 4096 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=alpine.qcow2,format=qcow2,if=virtio \
  -display gtk \
  -vga virtio
chmod +x ~/qemu-alpine/setup2.sh
./setup2.sh

# Inside the Alpine guest do

# Update available packages list and tools required to get filc-bootstrap
apk update
apk add git bash

# Copy the filc-bootstrap repo into the VM
git clone https://github.com/orsonteodoro/filc-bootstrap.git
cd filc-bootstrap

chmod +x ./setup-alpine-guest-filc-test.sh
./setup-alpine-guest-filc-test.sh
./bootstrap.sh --test-alpine
