# Instructions for creating a Alpine guest VM for testing bootstraping before the expensive Gentoo bootstrapping.
# This is for testing both the bootstrap scripts correctness quickly and musl build.

# WARNING:  Currently this testing case is on hold until the llvm annotations dependency can be resolved.
# Use the Debian testing instructions instead.

# Outside in the host do

mkdir -p ~/qemu-alpine
cd ~/qemu-alpine

# Get the latest stable Alpine
wget https://dl-cdn.alpinelinux.org/alpine/v3.23/releases/x86_64/alpine-standard-3.23.0-x86_64.iso

cd ~/qemu-alpine
qemu-img create -f raw alpine.raw 80G

# Do not change the -m values.  It may increase linking time.

# Contents of ~/qemu-alpine/setup.sh for distro install:
#!/bin/bash
qemu-system-x86_64 \
  -enable-kvm \
  -m 8192 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=alpine.raw,format=raw,if=virtio \
  -cdrom alpine-standard-3.23.0-x86_64.iso \
  -boot once=d \
  -display gtk \
  -vga virtio \
  "$@"
chmod +x ~/qemu-alpine/setup.sh
./setup.sh

# In the Alpine guest do

# login as root
setup-alpine
# Most options you just press enter.  The ones to worry about are
# Password → of your choosing
# Keyboard layout → usually us
# Disk → choose vda then choose sys mode
# Manually partition disks as follows if possible
#   vda
#     - 84.9 GB Primary xfs and with boot flag on placed at the beginning of the drive for /
#     - 1.0 GB Logical swap area
#
# * Do not add many more swap until we find out the real reason why.
#   Adding swap may increase linking time by adding more unnecessary caching.

# Install is done.

sync
# Shutdown the VM.

# Outside in the host do

# Contents of ~/qemu-alpine/setup2.sh to boot without ISO image:
#!/bin/bash
qemu-system-x86_64 \
  -enable-kvm \
  -m 8192 \
  -cpu host \
  -smp 4 \
  -nic user,model=virtio \
  -drive file=alpine.raw,format=raw,if=virtio \
  -display gtk \
  -vga virtio \
  "$@"
chmod +x ~/qemu-alpine/setup2.sh
./setup2.sh

# In the Alpine guest do

# Create an 12 GiB swap file
fallocate -l 12G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile

# Make it permanent across reboots
echo '/swapfile none swap sw 0 0' >> /etc/fstab

# Update available packages list and tools required to get filc-bootstrap
cd ~/
apk update
apk add git bash

# For developers/testers add screen for inspecting errors
apk add screen
# Increase scrollback
echo "defscrollback 50000" > ~/.screenrc
screen
# Use CTRL a [ to scroll back and to report/fix errors

# Copy the filc-bootstrap repo into the VM
git clone https://github.com/orsonteodoro/filc-bootstrap.git
cd filc-bootstrap

chmod +x ./setup-alpine-guest-filc-test.sh
./setup-alpine-guest-filc-test.sh
./bootstrap.sh --test-alpine
